import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-app.js"; import { getAuth, GoogleAuthProvider, signInWithPopup } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-auth.js"; import { getFirestore, collection, addDoc, onSnapshot, query, orderBy, serverTimestamp } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-firestore.js"; // 🌐 MESSAGE GLOBAL const WELCOME_MESSAGE = "Welcome to the academic and university forum"; // 🔧 CONFIG FIREBASE const firebaseConfig = { apiKey: "COLE_AQUI", authDomain: "COLE_AQUI", projectId: "COLE_AQUI", storageBucket: "COLE_AQUI", messagingSenderId: "COLE_AQUI", appId: "COLE_AQUI" }; // 🚀 INIT const app = initializeApp(firebaseConfig); const auth = getAuth(app); const db = getFirestore(app); // 🔐 Provider único (melhor prática) const provider = new GoogleAuthProvider(); let currentUser = null; // ====================== // 🔐 LOGIN GOOGLE // ====================== async function loginGoogle() { try { const result = await signInWithPopup(auth, provider); currentUser = result.user; const loginBox = document.getElementById("loginBox"); if (loginBox) loginBox.style.display = "none"; alert(`${WELCOME_MESSAGE}\n\nWelcome ${currentUser.displayName}`); } catch (error) { console.error("Login error:", error); alert("Login failed. Try again."); } } // ====================== // 📝 CRIAR POST // ====================== async function createPost() { const input = document.getElementById("postInput"); const text = input?.value?.trim(); if (!text) return; if (!currentUser) { alert("You must be logged in to post."); return; } try { await addDoc(collection(db, "posts"), { text, user: currentUser.displayName, photo: currentUser.photoURL || null, uid: currentUser.uid, createdAt: serverTimestamp() }); input.value = ""; } catch (error) { console.error("Error creating post:", error); } } // ====================== // 📡 REALTIME POSTS // ====================== const q = query(collection(db, "posts"), orderBy("createdAt", "desc")); onSnapshot(q, (snapshot) => { const container = document.getElementById("posts"); if (!container) return; container.innerHTML = ""; // 🔥 Welcome message fixo no topo const welcomeDiv = document.createElement("div"); welcomeDiv.className = "post"; welcomeDiv.innerHTML = `

${WELCOME_MESSAGE}

`; container.appendChild(welcomeDiv); snapshot.forEach((docSnap) => { const post = docSnap.data(); const div = document.createElement("div"); div.className = "post"; div.innerHTML = `

${post.user || "Unknown"}

${post.text || ""}

`; container.appendChild(div); }); }); // ====================== // 🌍 EXPORT GLOBAL (SEM POLUIR WINDOW) // ====================== window.loginGoogle = loginGoogle; window.createPost = createPost;
